	
1)	Найти самый частый вид применяемого наказания для каждого судьи:

        SELECT ID судьи, COUNT(ID вида наказания) AS count_наказание
	FROM Приговоры
	GROUP BY ID судьи 
	ORDER BY count_наказание DESC

2)	Найти средний срок приговора для разных категорий преступлений и порядкового номера осуждения (осудили в первый раз, второй, третий). Результат вывести в виде транспортированной таблицы (оси – категория, номер осуждения):

SELECT ID статьи AS категория преступления, [1] AS первый, [2] AS второй, [3] AS третий, [4] AS четверный
FROM (SELECT ID статьи.П1 AS категория преступления, count(ID преступника.2) AS count 
      FROM Приговоры.П1 
	       JOIN Приговоры П2  ON ID статьи.П1=ID статьи.П2
	  WHERE ID преступника.П1 = ID преступника.П2
	  GROUP BY ID статьи.П1
	  ) AS SourceTable
PIVOT 
       (AVG(срок приговора) 
       FOR категория преступления IN ([1], [2], [3], [4])
	   ) AS PivotTable
ORDER BY категория преступления
